export async function POST(req) {
    try {
      const data = await req.json(); // Await the promise to get the parsed data
      console.log(data); // Log the received data
  
      // Respond back to the client
      return new Response(JSON.stringify({ message: 'Data received successfully', data }), {
        status: 200,
        headers: {
          'Content-Type': 'application/json',
        },
      });
    } catch (error) {
      console.error('Error parsing JSON:', error);
      return new Response(JSON.stringify({ message: 'Error parsing data' }), {
        status: 400,
        headers: {
          'Content-Type': 'application/json',
        },
      });
    }
}
